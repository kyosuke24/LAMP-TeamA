DROP TABLE IF EXISTS Vor_table;
CREATE TABLE Vor_table (
Vor_id   INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
Vor_name VARCHAR(100)    NOT NULL DEFAULT '',
Vor_day DATE NOT NULL,
Vor_number INT DEFAULT 0,
Vor_place TEXT NOT NULL,
Vor_content TEXT NOT NULL,
Vor_explanation TEXT NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
